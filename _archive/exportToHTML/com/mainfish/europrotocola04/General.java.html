107      public void getDateTime () { 
108   
109          Calendar c = Calendar.getInstance(); 
110   
111          int day = c.get(Calendar.DATE); 
112          String dayX; 
113   
114          if (day < 10) { 
115              dayX = "0" + day; 
116          } else { 
117              dayX = String.valueOf(day); 
118          } 
119   
120          int month = (c.get(Calendar.MONTH)+ 1); 
121          String monthX; 
122   
123          if (month < 10) { 
124              monthX = "0" + month; 
125          } else { 
126              monthX = String.valueOf(month); 
127          } 
128   
129          int year = c.get(Calendar.YEAR); 
130          String yearX; 
131   
132          if (year < 10) { 
133              yearX = "0" + year; 
134          } else { 
135              yearX = String.valueOf(year); 
136          } 
137   
138          int hour = c.get(Calendar.HOUR_OF_DAY); 
139          String hourX; 
140   
141          if (hour < 10) { 
142              hourX = "0" + hour; 
143          } else { 
144              hourX = String.valueOf(hour); 
145          } 
146   
147          int minute = c.get(Calendar.MINUTE); 
148          String minuteX; 
149   
150          if (minute < 10) { 
151              minuteX = "0" + minute; 
152          } else { 
153              minuteX = String.valueOf(minute); 
154          } 
155   
156          gen_date_auto = dayX + "." + monthX + "." + yearX; 
157          gen_time_auto = hourX + ":" + minuteX; 
158   
159          gen_date = gen_date_auto; 
160          gen_time = gen_time_auto; 
161   
162      }
